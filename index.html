// ============================ // Balance - Google Apps Script // ============================ // Google Sheet 欄位： // A:日期 B:項目 C:金額 D:付款方式 E:幣別 F:分類 G:備註 H:匯率 I:台幣等值 // // 部署方式： // 1. 在 Google Sheets 點「擴充功能 → Apps Script」 // 2. 貼上這段程式碼（取代原本的） // 3. 把下面的 TOKEN 改成你自己的密碼 // 4. 部署 → 管理部署 → 鉛筆圖示 → 版本選「新版本」→ 部署 var TOKEN = "你自己設的密碼"; // ← 改成你的密碼 function doPost(e) { var data = JSON.parse(e.postData.contents); if (data.token !== TOKEN) { return ContentService.createTextOutput("Unauthorized"); } var sheet = SpreadsheetApp.getActiveSheet(); sheet.appendRow([ data.date, data.description, data.amount, data.payment, data.currency, data.category, "", // G: 備註（語音記帳不填） data.rate || 1, // H: 匯率 data.amountTWD || data.amount // I: 台幣等值 ]); return ContentService.createTextOutput("OK"); } function doGet(e) { var params = e.parameter; if (params.token !== TOKEN) { return ContentService.createTextOutput(JSON.stringify({ error: "Unauthorized" })) .setMimeType(ContentService.MimeType.JSON); } var sheet = SpreadsheetApp.getActiveSheet(); var data = sheet.getDataRange().getValues(); var result = []; for (var i = 1; i < data.length; i++) { result.push({ "日期": data[i][0] ? data[i][0].toString() : "", "項目": data[i][1] ? data[i][1].toString() : "", "金額": data[i][2] ? Number(data[i][2]) : 0, "付款方式": data[i][3] ? data[i][3].toString() : "", "幣別": data[i][4] ? data[i][4].toString() : "", "分類": data[i][5] ? data[i][5].toString() : "", "備註": data[i][6] ? data[i][6].toString() : "", "匯率": data[i][7] ? Number(data[i][7]) : 1, "台幣等值": data[i][8] ? Number(data[i][8]) : (data[i][2] ? Number(data[i][2]) : 0) }); } return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); }
